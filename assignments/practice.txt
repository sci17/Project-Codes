.data
	msg1: .asciiz "Enter number:"
	msg2: .asciiz "\n First Time: "
	
.globl main

main:
	li $v0, 4
	la $a0, msg1
	syscall
	
	li $v0, 5
	syscall
	move $t0, $v0
	
	li $v0,4
	la $a0, msg2
	
loop printing:
	beq $t0, $zero, exit
	addi $t0, $t0, -1
	syscall
	j loop_printing
	
exit:
	li $v0, 10
	syscall
